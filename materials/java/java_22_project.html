  • 객체지향 설계 부모(캐릭터) ➔ 자식(전사, 몬스터) 상속 구조 설계하기

  • 캡슐화/다형성 캐릭터의 체력을 안전하게 깎고, 직업마다 공격 방식 다르게 덮어쓰기

2. 💻 전체 소스 코드 (RPGMain.java)

VS Code에서 RPGMain.java 파일을 새로 만들고 아래 코드를 그대로 붙여넣기 한 뒤 실행(Run)해 보세요. 게임이 시작됩니다!

import java.util.Scanner;

// --- [1. 부모 클래스: 모든 생명체의 기본 뼈대] ---
class Character {
    String name;
    int hp;
    int atk; // 공격력
    
    // 생성자
    public Character(String name, int hp, int atk) {
        this.name = name;
        this.hp = hp;
        this.atk = atk;
    }
    
    // 공격 메서드 (자식들이 입맛대로 바꿀 예정)
    public void attack(Character target) {
        System.out.println(name + "의 일반 공격!");
        target.takeDamage(atk);
    }
    
    // 피해를 입는 메서드
    public void takeDamage(int damage) {
        this.hp -= damage;
        System.out.println(name + "이(가) " + damage + "의 피해를 입었습니다! [HP: " + Math.max(0, hp) + "]");
    }
    
    public boolean isDead() {
        return hp <= 0;
    }
}

// --- [2. 자식 클래스: 영웅 (전사)] ---
class Hero extends Character {
    public Hero(String name) {
        super(name, 100, 20); // 부모 생성자 호출 (내 체력 100, 공격력 20)
    }
    
    // 오버라이딩: 영웅만의 멋진 공격
    @Override
    public void attack(Character target) {
        System.out.println("🗡️ [" + name + "]의 강려크한 검술 공격!!");
        target.takeDamage(atk);
    }
    
    // 특수 스킬 (회복)
    public void heal() {
        hp += 30;
        System.out.println("✨ [" + name + "]이(가) 응급처치로 체력을 30 회복했습니다! [HP: " + hp + "]");
    }
}

// --- [3. 자식 클래스: 몬스터] ---
class Monster extends Character {
    public Monster(String name, int hp, int atk) {
        super(name, hp, atk);
    }
    
    @Override
    public void attack(Character target) {
        System.out.println("👺 [" + name + "]의 매서운 발톱 할퀴기!!");
        target.takeDamage(atk);
    }
}

// --- [4. 본부석 (실행 및 게임 루프)] ---
public class RPGMain {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        System.out.println("=== 🔥 용사의 마을에 오신 것을 환영합니다! 🔥 ===");
        System.out.print("당신의 영웅 이름을 정해주세요: ");
        String heroName = sc.nextLine();
        
        // 인스턴스(객체) 생성
        Hero myHero = new Hero(heroName);
        Monster goblin = new Monster("흉폭한 고블린", 60, 10);
        
        System.out.println("\n야생의 [" + goblin.name + "]이(가) 나타났습니다! 전투 시작!");
        
        // 무한 반복 전투 흐름 (둘 중 하나가 죽을 때까지)
        while (true) {
            System.out.println("\n----------------------------------------");
            System.out.println("나의 HP: " + myHero.hp + " | 상대 HP: " + goblin.hp);
            System.out.println("1. 기본 공격   2. 방어 후 체력회복");
            System.out.print("행동을 선택하세요: ");
            
            int choice = Integer.parseInt(sc.nextLine());
            
            // 1턴: 플레이어의 행동
            if (choice == 1) {
                myHero.attack(goblin); // 내가 몬스터를 때림
            } else if (choice == 2) {
                myHero.heal(); // 회복
            } else {
                System.out.println("잘못된 선택입니다. 턴이 날아갔습니다!");
            }
            
            // 몬스터 사망 체크 (break로 루프 탈출)
            if (goblin.isDead()) {
                System.out.println("\n🏆 축하합니다! 고블린을 물리쳤습니다!");
                break;
            }
            
            // 2턴: 몬스터의 반격
            System.out.println("");
            goblin.attack(myHero);
            
            // 플레이어 사망 체크
            if (myHero.isDead()) {
                System.out.println("\n💀 눈앞이 깜깜해졌습니다... 게임 오버.");
                break;
            }
        }
        
        System.out.println("게임이 종료되었습니다. 수고하셨습니다!");
        sc.close();
    }
}
