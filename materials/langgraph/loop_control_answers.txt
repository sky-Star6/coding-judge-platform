■ Q1. (실행 횟수 증가)
목표: For 방식에서 코드가 정확히 "10번" 돌고 종료되게 만들기
====================================================
import random
from typing import TypedDict
from langgraph.graph import StateGraph, END

class State(TypedDict):
    count: int      # 반복 횟수를 누적하는 변수
    max_loops: int  # 끝내기 위한 목표 횟수 변수

# 노드: 실행될 때마다 카운트를 +1 합니다.
def counter_node(state: State):
    current_count = state.get("count", 0)
    print(f"현재 횟수: {current_count + 1} / {state['max_loops']}")
    return {"count": current_count + 1}  # State에 새로운 횟수를 저장!

# 라우터 함수: 횟수에 도달했는지 검사
def check_count(state: State):
    if state["count"] < state["max_loops"]:
        return "loop"
    else:
        print("✅ 목표 횟수 도달. 스탬프 완성!")
        return "end"

builder = StateGraph(State)
builder.add_node("counter_node", counter_node)
builder.set_entry_point("counter_node")

# 반복 판단 연결 다리
builder.add_conditional_edges(
    "counter_node", 
    check_count, 
    {"loop": "counter_node", "end": END}
)

app = builder.compile()
if __name__ == "__main__":
    # 💡 정답 포인트: max_loops라는 목표치를 10으로 올려줍니다.
    app.invoke({"count": 0, "max_loops": 10})

■ Q2. (While 만족 점수 상향)
목표: 합격 커트라인이 80점에서 95점 이상으로 상향되었을 때의 라우터 처리
====================================================
import random
from typing import TypedDict
from langgraph.graph import StateGraph, END

class State(TypedDict):
    score: int

# 노드: 시험을 치고 랜덤으로 점수를 획득
def take_test_node(state: State):
    new_score = random.randint(50, 100) # 50~100 사이 랜덤 점수
    print(f"📝 시험 결과: {new_score}점 획득!")
    return {"score": new_score}

# 라우터 함수: 만족 점수에 도달했는지 판단
def router(state: State):
    # 💡 정답 포인트: 커트라인을 95점 이상으로 설정
    if state["score"] >= 95:  
        print("✅ 95점 이상! 합격입니다!")
        return "end"
    else:
        print("❌ 점수가 너무 낮습니다. 다시 시도!")
        return "retry"

builder = StateGraph(State)
builder.add_node("take_test", take_test_node)
builder.set_entry_point("take_test")

builder.add_conditional_edges(
    "take_test", 
    router, 
    {"retry": "take_test", "end": END}
)

app = builder.compile()
if __name__ == "__main__":
    app.invoke({"score": 0})

■ Q4. (상태 클래스 추가)
목표: State에 만족 여부를 뜻하는 Bool 변수(is_satisfactory) 추가
====================================================
import random
from typing import TypedDict
from langgraph.graph import StateGraph, END

# 💡 정답 포인트: State 구조에 bool 타입 추가
class State(TypedDict):
    count: int
    score: int
    is_satisfactory: bool  # 합격 여부를 판단할 스위치!

def test_node(state: State):
    current_count = state.get("count", 0)
    new_score = random.randint(0, 100)
    
    # 80점 이상이면 만족(True) 스위치를 켭니다.
    is_pass = True if new_score >= 80 else False
    
    print(f"[시도 {current_count + 1}] 점수: {new_score} / 만족 스위치: {is_pass}")
    
    return {
        "count": current_count + 1, 
        "score": new_score, 
        "is_satisfactory": is_pass
    }

def check_satisfactory(state: State):
    # 💡 Boolean 변수 자체를 조건문으로 활용
    if state["is_satisfactory"]:
        print("✅ 만족스러운 점수입니다. 루프를 종료합니다!")
        return "end"
    else:
        print("🔄 불만족. 다시 테스트합니다.")
        return "loop"

builder = StateGraph(State)
builder.add_node("test_node", test_node)
builder.set_entry_point("test_node")

builder.add_conditional_edges(
    "test_node", 
    check_satisfactory, 
    {"loop": "test_node", "end": END}
)

app = builder.compile()
if __name__ == "__main__":
    # 초기 상태에 is_satisfactory 기본값을 False로 설정하여 시작합니다.
    app.invoke({"count": 0, "score": 0, "is_satisfactory": False})

■ Q5. (이중 조건 라우터)
목표: 횟수 5번 도달이거나 100점 돌파 시 둘 중 하나라도 만족하면 종료
====================================================
import random
from typing import TypedDict
from langgraph.graph import StateGraph, END

class State(TypedDict):
    count: int
    score: int

def random_test_node(state: State):
    current_count = state.get("count", 0)
    current_score = state.get("score", 0)
    
    # 매번 10~40점 사이를 획득하여 누적
    added_score = random.randint(10, 40) 
    
    print(f"[스탬프: {current_count + 1}/5] {added_score}점 추가 획득! (총점: {current_score + added_score})")
    
    return {
        "count": current_count + 1, 
        "score": current_score + added_score
    }

# 💡 정답 포인트: or 연산자를 사용하여 2중 조건 검사
def double_check_router(state: State):
    # 스탬프(카운트) 5번이 다 찼거나, 점수가 100점을 넘었을 경우
    if state["count"] >= 5 or state["score"] >= 100:
        if state["score"] >= 100:
            print("🎉 [종료] 100점 돌파 달성!!")
        else:
            print("✅ [종료] 스탬프 5번을 모두 모았습니다.")
        return "end"
    else:
        print("🔄 아직 조건 미달. 계속 진행합니다.")
        return "loop"

builder = StateGraph(State)
builder.add_node("test_node", random_test_node)
builder.set_entry_point("test_node")

builder.add_conditional_edges(
    "test_node", 
    double_check_router, 
    {"loop": "test_node", "end": END}
)

app = builder.compile()
if __name__ == "__main__":
    app.invoke({"count": 0, "score": 0})
