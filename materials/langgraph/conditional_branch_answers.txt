■ Q1. (종료 조건 변경)
목표: `state["num"]`이 50이 아니라 100 이상일 때 끝나도록 조건문 변경
====================================================
from typing import TypedDict
from langgraph.graph import StateGraph, END

class State(TypedDict):
    num: int

def add_ten(state: State):
    current = state.get("num", 0)
    print(f"현재 숫자: {current} -> 10을 더합니다.")
    return {"num": current + 10}

def should_continue(state: State):
    # 💡 정답 포인트: 부등호 우측 숫자를 100으로 변경!
    if state["num"] >= 100:     
        print("✅ 100 돌파! 종료합니다.")
        return "end"     
    else:
        return "continue" 

builder = StateGraph(State)
builder.add_node("add_ten", add_ten)
builder.set_entry_point("add_ten")

builder.add_conditional_edges(
    "add_ten",
    should_continue,
    {
        "end": END,
        "continue": "add_ten"
    }
)
app = builder.compile()

if __name__ == "__main__":
    app.invoke({"num": 0})

■ Q4. (홀짝 분기 경로)
목표: 나머지 연산자를 사용해 짝수 경로와 홀수 경로로 나누는 조건 코딩
====================================================
from typing import TypedDict
from langgraph.graph import StateGraph, END
import random

class State(TypedDict):
    num: int

def generate_number(state: State):
    new_num = random.randint(1, 100)
    print(f"\n🎲 뽑힌 숫자: {new_num}")
    return {"num": new_num}

def process_even(state: State):
    print(f"➡️ [짝수 경로 실행] {state['num']}은(는) 짝수입니다.")
    return {}

def process_odd(state: State):
    print(f"➡️ [홀수 경로 실행] {state['num']}은(는) 홀수입니다.")
    return {}

# 💡 정답 포인트: 2로 나눈 나머지를 통한 분기
def check_even_odd(state: State):
    if state["num"] % 2 == 0:
        return "even_path"  # 짝수일 때 갈 방향
    else:
        return "odd_path"   # 홀수일 때 갈 방향

builder = StateGraph(State)
builder.add_node("generate_number", generate_number)
builder.add_node("process_even", process_even)
builder.add_node("process_odd", process_odd)

builder.set_entry_point("generate_number")

builder.add_conditional_edges(
    "generate_number",
    check_even_odd,
    {
        "even_path": "process_even",
        "odd_path": "process_odd"
    }
)

builder.add_edge("process_even", END)
builder.add_edge("process_odd", END)

app = builder.compile()

if __name__ == "__main__":
    app.invoke({"num": 0})

■ Q5. (3단 분기)
목표: 90점 이상, 50점 이상, 그 미만 3갈래로 분리되게 `elif`가 들어간 라우터 작성
====================================================
from typing import TypedDict
from langgraph.graph import StateGraph, END
import random

class State(TypedDict):
    num: int

def take_exam(state: State):
    score = random.randint(30, 100)
    print(f"\n📝 시험 결과: {score}점")
    return {"num": score}

def get_jackpot(state: State):
    print("🏆 [90점 이상] 훌륭합니다! 장학금을 받습니다.")
    return {}

def get_middle(state: State):
    print("👍 [50점 이상] 합격입니다. 다음 단계로 넘어갑니다.")
    return {}

def get_effort(state: State):
    print("💪 [50점 미만] 아쉽습니다. 재시험이 필요합니다.")
    return {}

# 💡 정답 포인트: elif를 활용한 3단 라우터
def score_router(state: State):
    score = state["num"]
    if score >= 90:
        return "jackpot"   # 90점 이상
    elif score >= 50:
        return "middle"    # 50점~89점
    else:
        return "effort"    # 50점 미만

builder = StateGraph(State)
builder.add_node("take_exam", take_exam)
builder.add_node("get_jackpot", get_jackpot)
builder.add_node("get_middle", get_middle)
builder.add_node("get_effort", get_effort)

builder.set_entry_point("take_exam")

builder.add_conditional_edges(
    "take_exam",
    score_router,
    {
        "jackpot": "get_jackpot",
        "middle": "get_middle",
        "effort": "get_effort"
    }
)

builder.add_edge("get_jackpot", END)
builder.add_edge("get_middle", END)
builder.add_edge("get_effort", END)

app = builder.compile()

if __name__ == "__main__":
    app.invoke({"num": 0})
